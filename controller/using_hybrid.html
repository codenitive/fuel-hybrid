    Controller Using Hybrid

    Controller_Hybrid is the easier way to combine both the functionality of Controller_Template and Controller_Rest using the same controller.

    All template feature available in Controller_Template is available in Crontroller_Hybrid.

    Example

    Here's an example of Controller_Hybrid.

    <?php

class Controller_Login extends Hybrid\Controller_Hybrid
{
    public $template = 'normal';

    /**
     * Display Login page
     *
     * @link    /login/index
     */
    public function action_index()
    {
        return $this->response(array(
            'content' => \View::forge('login'),
        ), 200);
    }

    /**
     * REST Login a user
     *
     * @link    POST /login/index.json
     */
    public function post_index()
    {
        $username = Input::post('username', '');
        $password = Input::post('password', '');

        // try to login the user

        return $this->response(array(
            'text' => 'User logged in'
        ), 200);
    }

     * @link    DELETE /login/index.json

    }

}
